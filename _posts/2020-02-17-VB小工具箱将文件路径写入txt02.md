为工具箱设置窗口靠边自动隐藏的功能

添加一个两个个command控件  

command1.caption = 新增
command2.caption = 读取

通过单击新增按钮选择文件路径并输出到123.txt中

通过读取按钮读取一行数据



Private Sub Command1_Click()
With CommonDialog1
.DialogTitle = "请选择文件"
.Filter = "*.*|*.*"
.CancelError = False
.ShowOpen
Text1.Text = .FileName
Open App.Path & "\123.txt" For Append As #1
Print #1, Text1.Text '把a的值写入123.txt
Close #1
End With
End Sub

Private Sub Form_Load()
Me.OLEDropMode = 1
End Sub



Private Sub Command2_Click()
Open App.Path & "\123.txt" For Input As #1 '备注：App.Path & "\123.txt"是程序目录下的123.txt
Dim Lines As String
Line Input #1, Lines '读取文本内容
Close #1
Text1.Text = Lines
End Sub
